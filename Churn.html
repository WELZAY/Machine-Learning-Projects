######Profit calculation

testnew<-test
testnew$si2014<-as.numeric(as.character(testnew$si2014))

testnew$dt_prediction<-unlist(result_b$prediction)
testnew$bdt_prediction<-unlist(result_boost$prediction)

#each new column contains the ranks of the corresponding prediction columns
testnew<- testnew %>% mutate(dt_rank = rank(desc(dt_prediction), ties.method = "first"))
testnew<- testnew %>% mutate(bdt_rank = rank(desc(bdt_prediction), ties.method = "first"))
###gamma here doesn't take into acccount costs yet
gamma<- mean(model_data$importo) 

cost_per_call <- 1
cost_per_email <- 0.15  

phone_response_rate <- 0.35
email_response_rate <- 0.15
threshold <- 0.5

total_budget <- 5000

# Proportions for calls and emails
ratio_calls <- 3
ratio_emails <- 2

summary(testnew)
#BUDGET 

# Calculate the total number of interactions based on the ratio and costs
total_interactions <- floor(total_budget / ((ratio_calls * cost_per_call) + (ratio_emails * cost_per_email)))

# Calculate the number of calls and emails based on the ratio
num_calls <- total_interactions * ratio_calls
num_emails <- total_interactions * ratio_emails

# Calculate the budget used for calls and emails
budget_for_calls <- num_calls * cost_per_call
budget_for_emails <- num_emails * cost_per_email

# Check if the allocated budget exceeds the total budget
if((budget_for_calls + budget_for_emails) > total_budget) {
  # Recalculate the number of calls and emails based on the remaining budget
  remaining_budget <- total_budget - budget_for_emails
  num_calls <- floor(remaining_budget / cost_per_call)
  budget_for_calls <- num_calls * cost_per_call
}
#Sorted customers

testnew <- testnew %>% 
  arrange(desc(bdt_rank))

# Calculate the potential profit from calls and emails
testnew_sorted <- testnew %>%
  mutate(
    bdt_profit_email = ifelse(bdt_prediction <= threshold & si2014 == 0, 
                              gamma * email_response_rate - cost_per_email, 
                              ifelse(bdt_prediction <= threshold & si2014 == 1, 
                                     -cost_per_email, 
                                     0)),
    bdt_profit_phone = ifelse(bdt_prediction <= threshold & si2014 == 0, 
                              gamma * phone_response_rate - cost_per_call, 
                              ifelse(bdt_prediction <= threshold & si2014 == 1, 
                                     -cost_per_call, 
                                     0)),
    profit_bdt = bdt_profit_email + bdt_profit_phone
  )
##threshold for number of visits
